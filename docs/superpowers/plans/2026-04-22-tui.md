> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an interactive TUI (prompt_toolkit + Rich) to ci-agent so users can converse with the CI agent in real time, replacing the one-shot `analyze` workflow with a streaming chat interface.

**Architecture:** A new `src/ci_optimizer/tui/` package handles all TUI concerns (repo detection, REPL loop, streaming render, slash commands, write confirmation panels). The existing `agents/`, `api/`, and `cli.py` are untouched except for wiring a new `chat` subcommand entry point. Each user message triggers a fresh `claude_agent_sdk.query()` call with the repo context embedded in the prompt; the async generator feeds a `StreamRenderer` that prints live to the terminal via Rich.

**Tech Stack:** `prompt_toolkit>=3.0` (input/history/key bindings), `rich>=13.0` (streaming markdown, panels, spinners), `claude_agent_sdk` (existing), `asyncio`.

"""Repo detection and user confirmation for TUI startup."""

from __future__ import annotations

import re
import subprocess
from dataclasses import dataclass
from pathlib import Path


@dataclass
class RepoContext:
    """Detected repository information."""

    local_path: Path
    owner: str | None
    repo: str | None
    branch: str | None

    @property
    def display_name(self) -> str:
        if self.owner and self.repo:
            return f"{self.owner}/{self.repo}"
        return str(self.local_path)


def parse_remote_url(url: str) -> tuple[str | None, str | None]:
    """Extract (owner, repo) from a git remote URL.

    Supports:
    - https://github.com/owner/repo.git
    - git@github.com:owner/repo.git
    """
    # HTTPS
    m = re.match(r"https?://[^/]+/([^/]+)/([^/]+?)(?:\.git)?$", url)
    if m:
        return m.group(1), m.group(2)
    # SSH
    m = re.match(r"git@[^:]+:([^/]+)/([^/]+?)(?:\.git)?$", url)
    if m:
        return m.group(1), m.group(2)
    return None, None


def _run(cmd: list[str], cwd: Path) -> str:
    """Run a git command, return stdout stripped, raise on error."""
    result = subprocess.run(cmd, cwd=cwd, capture_output=True, text=True)
    if result.returncode != 0:
        raise ValueError(result.stderr.strip() or f"command failed: {cmd}")
    return result.stdout.strip()


def detect_repo(path: Path) -> RepoContext:
    """Detect git repo at path. Raises ValueError if not a git repo."""
    try:
        root = Path(_run(["git", "rev-parse", "--show-toplevel"], cwd=path))
    except ValueError:
        raise ValueError(f"{path} is not a git repository")

    # Branch
    try:
        branch = _run(["git", "rev-parse", "--abbrev-ref", "HEAD"], cwd=root)
    except ValueError:
        branch = None

    # Remote URL → owner/repo
    owner, repo = None, None
    try:
        remote_url = _run(["git", "remote", "get-url", "origin"], cwd=root)
        owner, repo = parse_remote_url(remote_url)
    except ValueError:
        pass

    return RepoContext(local_path=root, owner=owner, repo=repo, branch=branch)
```

"""Slash command parser and handler registry for the TUI REPL."""

from __future__ import annotations

# Registry: name → help text
COMMANDS: dict[str, str] = {
    "help": "Show this help message",
    "clear": "Clear conversation history",
    "cost": "Show session token usage and cost",
    "repo": "Switch working repository: /repo [path]",
    "skills": "List active skills",
    "model": "Switch model: /model [model-name]",
}


def parse_command(text: str) -> tuple[str, list[str]] | None:
    """Parse a slash command from user input.

    Returns (command_name, args) if the text starts with '/' and the command
    is known, otherwise None.
    """
    text = text.strip()
    if not text.startswith("/"):
        return None
    parts = text[1:].split()
    if not parts:
        return None
    name = parts[0].lower()
    if name not in COMMANDS:
        return None
    return name, parts[1:]


def format_help() -> str:
    """Return a formatted help string for all commands."""
    lines = ["Available commands:"]
    for name, desc in COMMANDS.items():
        lines.append(f"  /{name:<10} {desc}")
    return "\n".join(lines)
```

"""Write confirmation panel and diff view for TUI."""

from __future__ import annotations

from dataclasses import dataclass
from enum import Enum


class ConfirmResult(Enum):
    CONFIRM = "confirm"       # y — execute all actions
    CANCEL = "cancel"         # n — abort
    DIFF = "diff"             # d — show diff then re-prompt
    FILES_ONLY = "files_only" # e — apply files only, no commit/PR


@dataclass
class WriteAction:
    """Describes a proposed write operation that needs user confirmation."""

    files: list[str]
    commit_message: str
    pr_branch: str | None
    pr_target: str | None
    diff: str

    @property
    def has_pr(self) -> bool:
        return bool(self.pr_branch and self.pr_target)


def parse_confirm_key(key: str) -> ConfirmResult | None:
    """Map a single keypress to a ConfirmResult. Returns None for unknown keys."""
    return {
        "y": ConfirmResult.CONFIRM,
        "Y": ConfirmResult.CONFIRM,
        "n": ConfirmResult.CANCEL,
        "N": ConfirmResult.CANCEL,
        "d": ConfirmResult.DIFF,
        "D": ConfirmResult.DIFF,
        "e": ConfirmResult.FILES_ONLY,
        "E": ConfirmResult.FILES_ONLY,
    }.get(key)


def render_confirmation_panel(action: WriteAction) -> str:
    """Return a Rich-markup string for the confirmation panel."""
    lines = [
        "[bold red]⚠  即将执行以下操作[/bold red]",
        "",
    ]
    if action.files:
        lines.append("[dim]📝 修改文件[/dim]")
        for f in action.files:
            lines.append(f"   [cyan]{f}[/cyan]")
    if action.commit_message:
        lines.append("[dim]📦 Git commit[/dim]")
        lines.append(f"   {action.commit_message}")
    if action.has_pr:
        lines.append("[dim]🔀 Pull Request[/dim]")
        lines.append(f"   [yellow]{action.pr_branch}[/yellow] → {action.pr_target}")
    lines += [
        "",
        "[green]\\[y][/green] 确认执行  "
        "[red]\\[n][/red] 取消  "
        "[yellow]\\[d][/yellow] 查看 diff  "
        "[blue]\\[e][/blue] 只修改文件，不提 PR",
    ]
    return "\n".join(lines)


async def prompt_confirmation(action: WriteAction) -> ConfirmResult:
    """Show confirmation panel and read user choice. Returns ConfirmResult."""
    from rich.console import Console
    from rich.panel import Panel
    from rich.markup import escape
    from prompt_toolkit import prompt as pt_prompt

    console = Console()

    while True:
        console.print(
            Panel(
                render_confirmation_panel(action),
                border_style="red",
                padding=(1, 2),
            )
        )

        if action.diff:
            raw = await _async_prompt("选择 [y/n/d/e]: ")
        else:
            raw = await _async_prompt("选择 [y/n/e]: ")

        result = parse_confirm_key(raw.strip())
        if result is None:
            console.print("[dim]无效输入，请输入 y / n / d / e[/dim]")
            continue

        if result == ConfirmResult.DIFF and action.diff:
            from rich.syntax import Syntax
            console.print(Syntax(action.diff, "diff", theme="monokai"))
            continue

        return result


async def _async_prompt(message: str) -> str:
    """Run prompt_toolkit prompt in a thread-safe async way."""
    import asyncio
    from prompt_toolkit import PromptSession
    session = PromptSession()
    loop = asyncio.get_event_loop()
    return await loop.run_in_executor(None, lambda: session.prompt(message))
```

"""Rich streaming renderer for claude_agent_sdk message stream."""

from __future__ import annotations

import asyncio
from dataclasses import dataclass, field
from typing import AsyncIterator, Any


@dataclass
class RenderResult:
    """Collected output from a streaming render pass."""

    text: str = ""
    cost_usd: float = 0.0
    duration_ms: int = 0


class StreamRenderer:
    """Consumes an async stream of claude_agent_sdk messages and renders to terminal.

    Args:
        quiet: If True, suppress all Rich output (for testing).
    """

    def __init__(self, quiet: bool = False) -> None:
        self._quiet = quiet

    async def stream(self, message_gen: AsyncIterator[Any]) -> RenderResult:
        """Consume message_gen and render to terminal. Returns collected result."""
        from claude_agent_sdk import AssistantMessage, TextBlock, ToolUseBlock, ResultMessage

        result = RenderResult()
        text_parts: list[str] = []

        if not self._quiet:
            from rich.console import Console
            from rich.live import Live
            from rich.spinner import Spinner
            from rich.text import Text
            console = Console()
        else:
            console = None

        async for message in message_gen:
            if isinstance(message, AssistantMessage):
                for block in message.content:
                    if isinstance(block, TextBlock):
                        text_parts.append(block.text)
                        if console:
                            from rich.markdown import Markdown
                            console.print(Markdown(block.text))
                    elif isinstance(block, ToolUseBlock):
                        if console:
                            console.print(
                                f"  [dim]⠸ {block.name}[/dim]",
                                highlight=False,
                            )
            elif isinstance(message, ResultMessage):
                result.cost_usd = message.total_cost_usd or 0.0
                result.duration_ms = message.duration_ms or 0
                if console and result.cost_usd:
                    console.print(
                        f"[dim]用时 {result.duration_ms / 1000:.1f}s · "
                        f"花费 ${result.cost_usd:.4f}[/dim]"
                    )

        result.text = "\n".join(text_parts)
        return result
```

"""prompt_toolkit PromptSession factory for the TUI REPL."""

from __future__ import annotations

from pathlib import Path

from prompt_toolkit import PromptSession
from prompt_toolkit.history import FileHistory
from prompt_toolkit.key_binding import KeyBindings
from prompt_toolkit.completion import WordCompleter
from prompt_toolkit.styles import Style

from ci_optimizer.tui.commands import COMMANDS

_HISTORY_FILE = Path.home() / ".ci-agent" / "history"

_STYLE = Style.from_dict({
    "prompt": "ansigreen bold",
})

_COMPLETER = WordCompleter(
    [f"/{name}" for name in COMMANDS],
    pattern=r"/\w*",
    sentence=True,
)


def build_session() -> PromptSession:
    """Build a configured PromptSession with history, completions, and key bindings."""
    _HISTORY_FILE.parent.mkdir(parents=True, exist_ok=True)

    kb = KeyBindings()

    @kb.add("c-c")
    def _handle_ctrl_c(event):
        """Ctrl+C cancels current input (does not exit)."""
        event.app.current_buffer.reset()
        event.app.output.write("\n")

    return PromptSession(
        history=FileHistory(str(_HISTORY_FILE)),
        completer=_COMPLETER,
        key_bindings=kb,
        style=_STYLE,
        complete_while_typing=False,
    )
```

def _confirm_repo(start_path: Path) -> RepoContext:
    """Detect and confirm repo with user. Loops until confirmed or manually entered."""
    while True:
        try:
            ctx = detect_repo(start_path)
            console.print(
                f"[yellow]? 检测到 Git 仓库：[/yellow][bold]{ctx.display_name}[/bold]"
            )
            if ctx.branch:
                console.print(f"[dim]  分支：{ctx.branch}[/dim]")
            answer = input("  使用此仓库？[Y]es / [n]o: ").strip().lower()
            if answer in ("", "y", "yes"):
                return ctx
        except ValueError as e:
            console.print(f"[red]{e}[/red]")

        manual = input("  请输入仓库路径或 GitHub owner/repo: ").strip()
        if not manual:
            continue
        # Handle owner/repo format
        if "/" in manual and not manual.startswith("/"):
            from ci_optimizer.resolver import resolve_input
            try:
                resolved = resolve_input(manual)
                start_path = resolved.local_path
            except Exception as exc:
                console.print(f"[red]无法解析：{exc}[/red]")
                continue
        else:
            start_path = Path(manual).expanduser().resolve()


def _build_chat_prompt(ctx: RepoContext, user_message: str) -> str:
    """Build the prompt sent to the agent for each user turn."""
    return (
        f"You are an expert CI/CD assistant working on the repository: {ctx.display_name}\n"
        f"Local path: {ctx.local_path}\n\n"
        f"User request: {user_message}\n\n"
        "Analyze the CI configuration and history as needed. "
        "For any proposed file changes, describe them clearly with file paths and diffs. "
        "Be concise and actionable."
    )


async def _handle_command(
    name: str,
    args: list[str],
    ctx: RepoContext,
    config: AgentConfig,
    session_cost: list[float],
) -> RepoContext:
    """Handle a slash command. Returns (possibly updated) RepoContext."""
    if name == "help":
        console.print(format_help())

    elif name == "clear":
        console.clear()
        _print_banner(ctx, config)

    elif name == "cost":
        console.print(f"[dim]本次会话花费：${sum(session_cost):.4f}[/dim]")

    elif name == "skills":
        from ci_optimizer.agents.skill_registry import get_registry
        skills = get_registry().get_active_skills()
        if skills:
            for s in skills:
                console.print(f"  [green]●[/green] {s.name}  [dim]{s.dimension}[/dim]")
        else:
            console.print("[dim]无活跃 skill[/dim]")

    elif name == "model":
        if args:
            config.model = args[0]
            console.print(f"[dim]已切换模型：{config.model}[/dim]")
        else:
            console.print(f"[dim]当前模型：{config.model}[/dim]")

    elif name == "repo":
        new_path = Path(args[0]).expanduser().resolve() if args else Path.cwd()
        try:
            ctx = detect_repo(new_path)
            console.print(f"[dim]已切换仓库：{ctx.display_name}[/dim]")
        except ValueError as e:
            console.print(f"[red]{e}[/red]")

    return ctx


async def run_tui(repo_path: Path | None = None) -> None:
    """Main TUI entry point."""
    from claude_agent_sdk import ClaudeAgentOptions, query

    config = AgentConfig.load()
    if not config.get_api_key():
        console.print("[red]Error: No API key configured. Run: ci-agent config set anthropic_api_key sk-ant-...[/red]")
        return

    start_path = repo_path or Path.cwd()
    ctx = _confirm_repo(start_path)

    _print_banner(ctx, config)

    session = build_session()
    renderer = StreamRenderer()
    session_cost: list[float] = []

    while True:
        try:
            user_input = await session.prompt_async("› ")
        except (EOFError, KeyboardInterrupt):
            console.print("\n[dim]Goodbye![/dim]")
            break

        user_input = user_input.strip()
        if not user_input:
            continue

        # Slash commands
        parsed = parse_command(user_input)
        if parsed is not None:
            name, args = parsed
            ctx = await _handle_command(name, args, ctx, config, session_cost)
            continue

        # LLM call
        prompt = _build_chat_prompt(ctx, user_input)
        options = ClaudeAgentOptions(
            system_prompt=(
                "You are an expert CI/CD assistant. Analyze CI configurations, diagnose "
                "failures, and suggest concrete fixes. Be concise and actionable."
            ),
            allowed_tools=["Read", "Glob", "Grep"],
            cwd=str(ctx.local_path),
            max_turns=config.max_turns,
        )
        if config.model:
            options.model = config.model

        try:
            result = await renderer.stream(query(prompt=prompt, options=options))
            session_cost.append(result.cost_usd)
        except Exception as e:
            console.print(f"[red]Error: {e}[/red]")
```
